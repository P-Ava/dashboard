let dataLokalUangPribadi = []; let modeEditAktif = false; // FUNGSI AMBIL DATA (URUT TANGGAL TERBARU) async function ambilDataUangPribadi() { try { const { data, error } = await supaClient .from('uang_pribadi') .select('*') .order('date', { ascending: false }); if (error) throw error; dataLokalUangPribadi = data || []; renderTabel(dataLokalUangPribadi); } catch (err) { console.error('Gagal mengambil data:', err.message); document.getElementById('tabel-body-supa').innerHTML = `Koneksi Supabase Gagal.`; } } // FUNGSI RENDER TABEL function renderTabel(dataArray) { const tbody = document.getElementById('tabel-body-supa'); tbody.innerHTML = ''; let akumulasiSaldo = 0; dataLokalUangPribadi.forEach(row => { const harga = row.harga || 0; const jumlah = row.jumlah || 1; if (row.transaksi === true) { akumulasiSaldo += (harga * jumlah); } else { akumulasiSaldo -= (harga * jumlah); } }); document.getElementById('total-saldo-pribadi').innerText = new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', maximumFractionDigits: 0 }).format(akumulasiSaldo); const thAksi = document.getElementById('th-aksi'); if (modeEditAktif) { thAksi.classList.remove('hidden'); } else { thAksi.classList.add('hidden'); } if (dataArray.length === 0) { tbody.innerHTML = `Tidak ada data yang cocok`; document.getElementById('keterangan-jumlah-supa').innerText = 'Menampilkan 0 catatan'; return; } dataArray.forEach((row, index) => { // KEMBALI MENGGUNAKAN HURUF KECIL 'id' SESUAI DATABASE const idRow = row.id; const namaBarang = row.nama_barang || 'Tanpa Nama'; const jumlah = row.jumlah || 1; const harga = row.harga || 0; const tanggal = row.date || '-'; const apakahPemasukan = row.transaksi === true; const formatHarga = new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', maximumFractionDigits: 0 }).format(harga); const tr = document.createElement('tr'); tr.className = `hover:bg-stone-50/50 transition ${index >= 10 ? 'hidden' : ''}`; let kolomAksiHtml = ''; if (modeEditAktif) { kolomAksiHtml = ` `; } tr.innerHTML = ` ${namaBarang} ${jumlah}x ${apakahPemasukan ? 'Pemasukan' : 'Penjualan'} ${apakahPemasukan ? '+' : '-'}${formatHarga} ${tanggal} ${kolomAksiHtml} `; tbody.appendChild(tr); }); const totalTampil = dataArray.length > 10 ? 10 : dataArray.length; document.getElementById('keterangan-jumlah-supa').innerText = `Menampilkan ${totalTampil} dari ${dataArray.length} catatan terdaftar`; } function toggleModeEdit() { modeEditAktif = !modeEditAktif; const btn = document.getElementById('btn-mode-edit'); if (modeEditAktif) { btn.classList.remove('text-stone-500', 'border-stone-200'); btn.className += ' bg-brand text-white border-brand'; } else { btn.classList.remove('bg-brand', 'text-white', 'border-brand'); btn.className += ' border-stone-200 text-stone-500'; } renderTabel(dataLokalUangPribadi); } function bukaModalTambah() { document.getElementById('form-id').value = ''; document.getElementById('modal-judul').innerText = 'Tambah Transaksi Baru'; document.getElementById('form-date').value = new Date().toISOString().split('T')[0]; document.getElementById('modal-transaksi').classList.remove('hidden'); } // DI SINI KUNCI PERBAIKANNYA: Menggunakan row.id (huruf kecil) function bukaModalEdit(id) { const dataRow = dataLokalUangPribadi.find(row => String(row.id) === String(id)); if (!dataRow) { console.error("Data ID tidak ditemukan lokal:", id); return; } document.getElementById('form-id').value = dataRow.id; // <-- Diisi dengan id huruf kecil document.getElementById('modal-judul').innerText = 'Edit Transaksi Lama'; document.getElementById('form-nama').value = dataRow.nama_barang || ''; document.getElementById('form-jumlah').value = dataRow.jumlah || 1; document.getElementById('form-transaksi').value = dataRow.transaksi === true ? 'true' : 'false'; document.getElementById('form-harga').value = dataRow.harga || 0; document.getElementById('form-date').value = dataRow.date || ''; document.getElementById('modal-transaksi').classList.remove('hidden'); } function tutupModal() { document.getElementById('form-tambah-supa').reset(); document.getElementById('modal-transaksi').classList.add('hidden'); } // SAVE / UPDATE HANDLER document.getElementById('form-tambah-supa').addEventListener('submit', async function(e) { e.preventDefault(); const id = document.getElementById('form-id').value; const nama_barang = document.getElementById('form-nama').value.trim(); const jumlah = parseInt(document.getElementById('form-jumlah').value); const transaksi = document.getElementById('form-transaksi').value === 'true'; const harga = parseFloat(document.getElementById('form-harga').value); const date = document.getElementById('form-date').value; try { if (id) { // MENGGUNAKAN 'id' HURUF KECIL UNTUK KUERI UPDATE KE SUPABASE const { error } = await supaClient .from('uang_pribadi') .update({ nama_barang, jumlah, transaksi, harga, date }) .eq('id', id); if (error) throw error; } else { const { error } = await supaClient .from('uang_pribadi') .insert([{ nama_barang, jumlah, transaksi, harga, date }]); if (error) throw error; } tutupModal(); ambilDataUangPribadi(); } catch (err) { alert('Gagal memproses data ke Supabase: ' + err.message); } }); // LIVE FILTER INPUT document.getElementById('input-cari-gabungan').addEventListener('input', function(e) { const kataKunci = e.target.value.toLowerCase().trim(); const dataHasilFilter = dataLokalUangPribadi.filter(row => { const namaBarang = (row.nama_barang || '').toLowerCase(); const tanggal = (row.date || '').toLowerCase(); return namaBarang.includes(kataKunci) || tanggal.includes(kataKunci); }); renderTabel(dataHasilFilter); }); window.addEventListener('DOMContentLoaded', ambilDataUangPribadi);
