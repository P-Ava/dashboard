// FUNGSI RENDER TABEL (PENSIL AKSIL SELALU MUNCUL) function renderTabel(dataArray) { const tbody = document.getElementById('tabel-body-supa'); tbody.innerHTML = ''; // Hitung akumulasi total saldo dari SELURUH data di Supabase let akumulasiSaldo = 0; dataLokalUangPribadi.forEach(row => { const harga = row.harga || 0; const jumlah = row.jumlah || 1; if (row.transaksi === true) { akumulasiSaldo += (harga * jumlah); } else { akumulasiSaldo -= (harga * jumlah); } }); document.getElementById('total-saldo-pribadi').innerText = new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', maximumFractionDigits: 0 }).format(akumulasiSaldo); // KOLOM AKSI SEKARANG SELALU DITAMPILKAN const thAksi = document.getElementById('th-aksi'); thAksi.classList.remove('hidden'); if (dataArray.length === 0) { tbody.innerHTML = `Tidak ada data yang cocok`; document.getElementById('keterangan-jumlah-supa').innerText = 'Menampilkan 0 catatan'; return; } dataArray.forEach((row, index) => { const id = row.id; const namaBarang = row.nama_barang || 'Tanpa Nama'; const jumlah = row.jumlah || 1; const harga = row.harga || 0; const tanggal = row.date || '-'; const apakahPemasukan = row.transaksi === true; const formatHarga = new Intl.NumberFormat('id-ID', { style: 'currency', currency: 'IDR', maximumFractionDigits: 0 }).format(harga); const tr = document.createElement('tr'); tr.className = 'hover:bg-stone-50/50 transition'; // Sembunyikan baris ke-11 dan seterusnya (maksimal tampil 10) if (index >= 10) { tr.className += ' hidden'; } // Tombol pensil aksi langsung dipasang tanpa syarat modeEditAktif let kolomAksiHtml = ` `; tr.innerHTML = ` ${namaBarang} ${jumlah}x ${apakahPemasukan ? 'Pemasukan' : 'Penjualan'} ${apakahPemasukan ? '+' : '-'}${formatHarga} ${tanggal} ${kolomAksiHtml} `; tbody.appendChild(tr); }); const totalTampil = dataArray.length > 10 ? 10 : dataArray.length; document.getElementById('keterangan-jumlah-supa').innerText = `Menampilkan ${totalTampil} dari ${dataArray.length} catatan terdaftar`; }
